        # Program to check if a number is prime or not in Python
        num = 29  # You can change this value to any positive integer

        # Define a flag variable
        flag = False

        if num == 1:
            print(num, "is not a prime number")
        elif num > 1:
            # Check for factors
            for i in range(2, num):
                if (num % i) == 0:
                    # If a factor is found, set the flag to True and break out of the loop
                    flag = True
                    break

            # Check if the flag is True
            if flag:
                print(num, "is not a prime number")
            else:
                print(num, "is a prime number")
